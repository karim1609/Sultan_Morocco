\section{Contrat JSON — Hôtels}

\textbf{Requête :} \texttt{GET /api/hotels?offset=0\&limit=30\&sort=best\_value}

\textbf{Réponse 200 :}
\begin{lstlisting}[language={},caption={Structure JSON}]
{
  "error": null,
  "result": {
    "total_count": 30,
    "limit": 30,
    "offset": 0,
    "list": [ { "name": "...", "geo": {...}, ... } ]
  }
}
\end{lstlisting}

\section{Contrat JSON — Restaurants}

\textbf{Paramètres :} \texttt{city}, \texttt{sort} $\in$ \{\texttt{rating}, \texttt{price\_asc}, \texttt{price\_desc}\}.

\section{Contrat JSON — Chatbot}

\textbf{Requête :} \texttt{POST /api/chat} body \texttt{\{ "message": "..." \}}

\textbf{Réponses types :}
\begin{itemize}
  \item \texttt{type: "text"} + \texttt{reply} + \texttt{options[]}
  \item \texttt{type: "buttons"}
  \item \texttt{type: "cards"} + \texttt{items[]} (nom, lieu, note, image)
\end{itemize}
