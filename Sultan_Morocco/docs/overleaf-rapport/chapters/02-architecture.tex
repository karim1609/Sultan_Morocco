\section{Principes directeurs}

\begin{itemize}
  \item \textbf{Separation des concerns} : MVC Laravel, API JSON decouples des vues Blade.
  \item \textbf{Fail-safe listings} : pas de single point of failure API tierce pour hotels/restaurants.
  \item \textbf{Progressive enhancement} : Nominatim enrichit l'UX sans bloquer le coeur metier.
  \item \textbf{Convention over configuration} : routes \texttt{web.php}, migrations, seeders standards.
\end{itemize}

\section{Decisions d'architecture (ADR synthese)}

\begin{infobox}{ADR-001 : Listings statiques en PHP}
\textbf{Contexte} : APIs hotelieres payantes / instables.\\
\textbf{Decision} : tableaux \texttt{\$HOTELS} et \texttt{\$RESTAURANTS} dans les controleurs.\\
\textbf{Consequences} : + simplicite, + performance ; -- mise a jour manuelle, pas de sync temps reel.
\end{infobox}

\begin{infobox}{ADR-002 : Session driver database}
\textbf{Contexte} : scalabilite legere multi-instances.\\
\textbf{Decision} : \texttt{SESSION\_DRIVER=database} + table \texttt{sessions}.\\
\textbf{Consequences} : sessions persistees ; necessite migrations a jour.
\end{infobox}

\begin{infobox}{ADR-003 : Messagerie relationnelle normalisee}
\textbf{Contexte} : remplacer ancienne table \texttt{chat\_messages} (bot).\\
\textbf{Decision} : \texttt{conversations} + pivot + \texttt{messages} avec \texttt{read\_at}.\\
\textbf{Consequences} : modele extensible vers groupes / pieces jointes en V2.
\end{infobox}
